#include<iostream>
#include<cstdlib>
using namespace std;
enum enGameOptions {Rock=1,Paper=2,Scissor=3};
enum enResult{Win=1,Lose=2,Draw=3};

int NumberOfRounds()
{
    int num;
    cout << "Please Enter Number Of Rounds you Want to Play : ";
    cin >> num;
    
    if (num < 0)
    {
        do {
            cout << endl;
            cout << "Please Enter A POSITIVE Number To Play : ";
            cin >> num;
        } while (num < 0);
    }
    return num;
}

enResult WhoWin(enGameOptions Player, enGameOptions Computer)
{
    if (Player == Computer)
        return Draw;

    switch (Player)
    {
    case Rock:
        return (Computer == Scissor) ? Win : Lose;

    case Paper:
        return (Computer == Rock) ? Win : Lose;

    case Scissor:
        return (Computer == Paper) ? Win : Lose;
    }

    return Draw;
}

void PrintWhoWin(enResult Result)
{
    if (Result == Win) {
        cout << " You Win !";
        system("color A1");

    }
    else if (Result == Lose)
    {
        cout << " You Lost :( ";
        system("color C0");
    }
    else
    {
        cout << " Draw !";
        system("color E0");
    }
}
int RandomNumber(int From, int To)
{
    int RandNum = rand() % (To - From + 1) + From;
    return RandNum;
};

enGameOptions RandomChoice()
{
    return (enGameOptions) RandomNumber(1, 3);
}
enGameOptions PlayerChoice()
{
    int Choice = 0;
    cout << "\nChoose Your Move:\n";
    cout << "[1] Rock\n";
    cout << "[2] Paper\n";
    cout << "[3] Scissor\n";
    cout << "Your Choice: ";
    cin >> Choice;
    if (!(Choice >= 1 && Choice <= 3))
    {
        do {
            cout << endl;
            cout << "Please Enter Number From 1 To 3: ";
            cin >> Choice;
        } while (!(Choice >= 1 && Choice <= 3));
    }
    
    return (enGameOptions)Choice;
    
}
void PrintChoice(enGameOptions Choice)
{
    if (Choice == Paper)
        cout << "Paper\n";
    else if (Choice == Rock)
        cout << "Rock\n";
    else if (Choice == Scissor)
        cout << "Scissor\n";
    else
        cout << " PLEASE CHOISE NUMBER FROM 1 TO 3 ! \n";
        


}
void FullRound(int &PlayerCounter,int &ComputerCounter)
{
    enGameOptions Player_Choice, Computer_Choice;

    
    cout << "\n------------------------------------------\n";
  

    Player_Choice = PlayerChoice();
    Computer_Choice = RandomChoice();

    cout << "\nPlayer Choice   : ";
    PrintChoice(Player_Choice);

    cout << "\nComputer Choice : ";
    PrintChoice(Computer_Choice);

    if (WhoWin(Player_Choice, Computer_Choice) == Win)
        PlayerCounter++;
    else if (WhoWin(Player_Choice, Computer_Choice) == Lose)
        ComputerCounter++;

    cout << "\n\nResult          : ";
    PrintWhoWin(WhoWin(Player_Choice, Computer_Choice));

    cout << "\n------------------------------------------\n";
}
void PrintFinalResult(int win, int lose)
{
    cout << "\n";
    cout << "==================================================\n";
    cout << "                  FINAL RESULT\n";
    cout << "==================================================\n\n";
    if (win > lose)
    {
        cout << "Player Score   : " << win << endl;
        cout << "Computer Score : " << lose << endl;
        cout << "------------------------------------------\n";
        cout << "Winner         : You\n";
        system("color A1");
        
    }
    else if (lose > win)
    {
        cout << "Player Score   : " << win << endl;
        cout << "Computer Score : " << lose << endl;
        cout << "------------------------------------------\n";
        cout << "Winner         : Computer\n";
        system("color C0");

    }
    else
    {
        cout << "Player Score   : " << win << endl;
        cout << "Computer Score : " << lose << endl;
        cout << "------------------------------------------\n";
        cout << "Winner         : Draw\n";
        system("color E0");
    }
    
}
void StartGame()
{
    bool Continue;

    do
    {
        cout << "\n";
        cout << "==================================================\n";
        cout << "           ROCK - PAPER - SCISSORS GAME\n";
        cout << "==================================================\n\n";

        int WinCont = 0, LoseCount = 0;

        int Number_Rounds = NumberOfRounds();

        for (int i = 1; i <= Number_Rounds; i++)
        {
            cout << "\n";
            cout << "==================================================\n";
            cout << "                 ROUND [" << i << "]\n";
            cout << "==================================================\n";
            FullRound(WinCont, LoseCount);
        }

        PrintFinalResult(WinCont, LoseCount);

        cout << "\n==================================================\n";
        cout << "Would you like to play again?\n";
        cout << "[1] Yes\n";
        cout << "[0] No\n";
        cout << "Choice: ";
        cin >> Continue;

        if (Continue)
        {
            system("cls");
        }

    } while (Continue);
}

int main()
{
    srand((unsigned)time(NULL));
    
    StartGame();

    return 0;
}
